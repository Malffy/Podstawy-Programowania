#include "pch.h"
#include <iostream>
#include <vector>
using namespace std;

void wypisz(int w)
{
	cout << w << "\n";
}

void wypisz(std::vector<int>tab)
{
	for (int i = 0;i < tab.size();i++)
	{
		std::cout << tab[i] << ", ";
	}
	std::cout<<std::endl;
}

bool CzyNależy(std::vector<int>tab, int element)
{
	for (int i = 0;i < tab.size();i++)
	{
		if (tab[i] == element)
		{
			return true;
		}
	}
	return false;
}


int suma(int a, int b)
{
	return a + b;
}

int suma(int a, int b, int c)
{
	int wynik = suma(a, b);
	return suma(wynik, c);
}


int main()
{
	std::vector<int>tab;
	tab.push_back(5);
	wypisz(tab[0]);
	tab = { 1,3,5 };
	std::cout << tab[0] << ", " << tab[1] << ", " << tab[2] << "\n";

	wypisz(tab);

	int a = 5;
	wypisz(a);

	int b = suma(a, 5, 3);
	wypisz(b);

	if (a > b)
	{
		wypisz(a);
	}
	else
	{
		std::cout << "Ale lipa!!!\n";
	}

	bool wynik = CzyNależy(tab, 4);
	if (wynik)
	{
		std::cout << "TAK" << std::endl;
	}
	else
	{
		std::cout << "Nie" << std::endl;
	}
	{


	}

	system("pause");
}
