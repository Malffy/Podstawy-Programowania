#include "pch.h"
#include <iostream>
#include <vector>

void wypisz(int w)
{
	std::cout << w << "\n";
}

int suma(int a, int b)
{
	return a + b;
}

int suma(int a, int b, int c)
{
	int wynik = suma(a, b);
	return suma(wynik,c);
}


int main()
{	
	std::vector<int>tab;
	tab.push_back(5);
	wypisz(tab[0]);

	int a = 5 + 3;
	wypisz(a);

	int b = suma(a,5,3);
	wypisz(b);

	if (a > b)
	{
		wypisz(a);
	}
	else
	{
		std::cout << "Ale lipa!!!\n";
	}

	system("pause");
}
